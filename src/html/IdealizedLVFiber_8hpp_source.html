123  //compute fiber parameters

125  const Scalar d_focal = 45.0;
126  const Scalar nu_endo = 0.6;
127  const Scalar nu_epi = 0.8;
128  const Scalar endo_r_1 = d_focal * std::sinh(nu_endo);
129  const Scalar endo_r_2 = d_focal * std::cosh(nu_endo);
130  const Scalar epi_r_1 = d_focal * std::sinh(nu_epi);
131  const Scalar epi_r_2 = d_focal * std::cosh(nu_epi);
132  const Scalar endo_epi_r_1_delta = std::abs(endo_r_1 - epi_r_1);
133  const Scalar endo_epi_r_2_delta = std::abs(endo_r_2 - epi_r_2);
134  const Scalar r_s = endo_r_1 + endo_epi_r_1_delta * t;
135  const Scalar r_e = endo_r_2 + endo_epi_r_2_delta * t;
136  const Scalar u_deg = std::acos(support_point[2] / r_e) * (180.0 / M_PI);
137  const Scalar u_rad = u_deg * (M_PI / 180.0);
138  const Scalar v_deg1 = std::asin(support_point[1] / (r_s * std::sin(u_rad))) * (180.0 / M_PI);
139  const Scalar v_deg2 = std::acos(support_point[0] / (r_s * std::sin(u_rad))) * (180.0 / M_PI);
140  const Scalar v_rad1 = v_deg1 * (M_PI / 180.0);
141  const Scalar v_rad2 = v_deg2 * (M_PI / 180.0);
142  const Scalar alpha_deg = 90.0 - 180.0 * t;
143  const Scalar alpha_rad = alpha_deg * (M_PI / 180.0);
144 
145  //sometimes when the argument of the std::asin or std::acos is ~1/~-1 we have detected numerical imprecisions that will lead to NaN.
146  //We are skipping those Dofs contributions (from tests we have detected only one point over ~74k total Dofs (with r = 1))
147  if ((std::isnan(v_rad1) && std::isnan(v_rad2)) || std::isnan(u_rad)) {

154  continue;
155  }
156  //angle v can be recovered by using 2 equations (https://pubmed.ncbi.nlm.nih.gov/26807042/)
157  const Scalar v_rad = std::isnan(v_rad1) ? v_rad2 : v_rad1;
158 
159  //declare partial derivative vectors (handmade derivative)
160  std::vector<Scalar> dx_du = {
161  r_s * std::cos(u_rad) * std::cos(v_rad),
162  r_s * std::cos(u_rad) * std::sin(v_rad),
163  -r_e * std::sin(u_rad)
164  };
165  std::vector<Scalar> dx_dv = {
166  -1.0 * r_s * std::sin(u_rad) * std::sin(v_rad),
167  r_s * std::sin(u_rad) * std::cos(v_rad),
168  0
169  };
170  normalize(dx_du, norm(dx_du));
171  normalize(dx_dv, norm(dx_dv));
172  //compute vector f
173  for (unsigned i=0; i<dim; ++i) {
174  f[i] += dx_du[i] * std::sin(alpha_rad) + dx_dv[i] * std::cos(alpha_rad);
175  }
176  }
177  out_tensor += (fiber_pressure * dealii::outer_product(f, f));
